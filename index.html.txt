    <script>
        // DOM Elements
        const inputs = document.querySelectorAll('#sig-form input');
        const previewDiv = document.getElementById('signature-preview');
        
        // Update signature function
        function updateSignature() {
            const name = document.getElementById('f-name').value || 'Name';
            const title = document.getElementById('f-title').value || 'Title';
            const phone = document.getElementById('f-phone').value || 'Phone';
            const email = document.getElementById('f-email').value || 'Email';
            const website = document.getElementById('f-website').value || 'Website';

            // HTML Structure for the Signature
            // Updated inline styles to use Aptos as the primary font
            const sigHTML = `
<table cellpadding="0" cellspacing="0" border="0" style="font-family: 'Aptos', Arial, Helvetica, sans-serif; font-size: 14px; color: #333333; line-height: 1.4;">
    <tr>
        <td style="padding-right: 20px; border-right: 2px solid #704EA0; text-align: center; vertical-align: middle;">
            <img src="https://via.placeholder.com/100x100.png?text=BPW+PNG+Logo" alt="BPW PNG Logo" width="100" style="display: block; max-width: 100px; border-radius: 50%;">
        </td>
        <td style="padding-left: 20px; vertical-align: middle;">
            <p style="margin: 0 0 5px 0; font-size: 18px; font-weight: bold; color: #704EA0;">${name}</p>
            <p style="margin: 0 0 10px 0; font-size: 14px; font-style: italic; color: #555555;">${title}<br><strong style="color: #333;">BPW PNG</strong></p>
            
            <table cellpadding="0" cellspacing="0" border="0" style="font-family: 'Aptos', Arial, Helvetica, sans-serif; font-size: 13px; color: #333333;">
                <tr style="${phone ? '' : 'display:none;'}">
                    <td style="padding-bottom: 3px;"><strong style="color: #704EA0;">P:</strong></td>
                    <td style="padding-bottom: 3px; padding-left: 5px;">${phone}</td>
                </tr>
                <tr style="${email ? '' : 'display:none;'}">
                    <td style="padding-bottom: 3px;"><strong style="color: #704EA0;">E:</strong></td>
                    <td style="padding-bottom: 3px; padding-left: 5px;"><a href="mailto:${email}" style="color: #333333; text-decoration: none;">${email}</a></td>
                </tr>
                <tr style="${website ? '' : 'display:none;'}">
                    <td><strong style="color: #704EA0;">W:</strong></td>
                    <td style="padding-left: 5px;"><a href="https://${website.replace(/^https?:\/\//,'')}" style="color: #333333; text-decoration: none;">${website}</a></td>
                </tr>
            </table>
        </td>
    </tr>
</table>
            `;

            previewDiv.innerHTML = sigHTML;
        }

        // Add event listeners to input fields
        inputs.forEach(input => {
            input.addEventListener('input', updateSignature);
        });

        // Copy functionality
        function copySignature() {
            const range = document.createRange();
            range.selectNode(previewDiv);
            const windowSelection = window.getSelection();
            windowSelection.removeAllRanges();
            windowSelection.addRange(range);
            
            try {
                document.execCommand('copy');
                const msg = document.getElementById('success-msg');
                msg.style.display = 'block';
                setTimeout(() => { msg.style.display = 'none'; }, 3000);
            } catch (err) {
                console.error('Failed to copy signature', err);
                alert("Failed to copy. Please manually highlight the signature and copy it.");
            }
            
            windowSelection.removeAllRanges();
        }

        // Initial render
        updateSignature();
    </script>
