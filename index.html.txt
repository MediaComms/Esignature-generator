    <script>
        const inputs = document.querySelectorAll('#sig-form input');
        const previewDiv = document.getElementById('signature-preview');
        
        function updateSignature() {
            const name = document.getElementById('f-name').value;
            const title = document.getElementById('f-title').value;
            const phone = document.getElementById('f-phone').value;
            const email = document.getElementById('f-email').value;
            const website = document.getElementById('f-website').value;

            // Using the filename of the image you uploaded to GitHub
            const logoUrl = window.location.href.substring(0, window.location.href.lastIndexOf('/') + 1) + "BPW Logo - PNG.jpg";

            const sigHTML = `
<table cellpadding="0" cellspacing="0" border="0" style="font-family: 'Aptos', Arial, sans-serif; font-size: 14px; color: #333333;">
    <tr>
        <td style="padding-right: 20px; border-right: 2px solid #704EA0; vertical-align: middle;">
            <img src="${logoUrl}" alt="BPW PNG" width="120" style="display: block;">
        </td>
        <td style="padding-left: 20px; vertical-align: middle;">
            <p style="margin: 0; font-size: 18px; font-weight: bold; color: #704EA0;">${name}</p>
            <p style="margin: 0 0 10px 0; color: #555555;">${title} | <strong>BPW PNG</strong></p>
            <p style="margin: 0; font-size: 13px;"><strong>P:</strong> ${phone}</p>
            <p style="margin: 0; font-size: 13px;"><strong>E:</strong> <a href="mailto:${email}" style="color: #333; text-decoration: none;">${email}</a></p>
            <p style="margin: 0; font-size: 13px;"><strong>W:</strong> <a href="http://${website}" style="color: #333; text-decoration: none;">${website}</a></p>
        </td>
    </tr>
</table>`;
            previewDiv.innerHTML = sigHTML;
        }

        inputs.forEach(input => input.addEventListener('input', updateSignature));
        
        function copySignature() {
            const range = document.createRange();
            range.selectNode(previewDiv);
            window.getSelection().removeAllRanges();
            window.getSelection().addRange(range);
            document.execCommand('copy');
            document.getElementById('success-msg').style.display = 'block';
        }

        updateSignature();
    </script>
